00149 struct MetaDataBlock
00150 {
00151     PVRTuint32  DevFOURCC;      //A 4cc descriptor of the data type's creator. Values equating to values between 'P' 'V' 'R' 0 and 'P' 'V' 'R' 255 will be used by our headers.
00152     PVRTuint32  u32Key;         //A DWORD (enum value) identifying the data type, and thus how to read it.
00153     PVRTuint32  u32DataSize;    //Size of the Data member.
00154     PVRTuint8*  Data;           //Data array, can be absolutely anything, the loader needs to know how to handle it based on DevFOURCC and Key. Use new operator to assign to it.
00155         
00156     /*!***********************************************************************
00157         @Function       MetaDataBlock
00158         @Description    Meta Data Block Constructor
00159     *************************************************************************/
00160     MetaDataBlock() : DevFOURCC(0), u32Key(0), u32DataSize(0), Data(NULL)
00161     {}
00162         
00163     /*!***********************************************************************
00164         @Function       MetaDataBlock
00165         @Description    Meta Data Block Copy Constructor
00166     *************************************************************************/
00167     MetaDataBlock(const MetaDataBlock& rhs)  : DevFOURCC(rhs.DevFOURCC), u32Key(rhs.u32Key), u32DataSize(rhs.u32DataSize)
00168     {
00169         //Copy the data across.
00170         Data = new PVRTuint8[u32DataSize];
00171         for (PVRTuint32 uiDataAmt=0; uiDataAmt<u32DataSize; ++uiDataAmt)
00172         {
00173             Data[uiDataAmt]=rhs.Data[uiDataAmt];
00174         }
00175     }
00176 
00177     /*!***********************************************************************
00178         @Function       ~MetaDataBlock
00179         @Description    Meta Data Block Destructor
00180     *************************************************************************/
00181     ~MetaDataBlock()
00182     {
00183         if (Data) 
00184             delete [] Data;
00185         Data = NULL;
00186     }
00187 
00188     /*!***********************************************************************
00189         @Function       SizeOfBlock
00190         @Return         size_t Size (in a file) of the block.
00191         @Description    Returns the number of extra bytes this will add to any output files.
00192     *************************************************************************/
00193     size_t SizeOfBlock() const
00194     {
00195         return sizeof(DevFOURCC)+sizeof(u32Key)+sizeof(u32DataSize)+u32DataSize;
00196     }

00198     /*!***********************************************************************
00199         @Function       operator=
00200         @Return         MetaDataBlock This MetaDataBlock after the operation.
00201         @Description    Assigns one MetaDataBlock to the other.
00202     *************************************************************************/
00203     MetaDataBlock& operator=(const MetaDataBlock& rhs)
00204     {
00205         if (&rhs==this)
00206             return *this;
00207 
00208         //Remove pre-existing data.
00209         if (Data)
00210             delete [] Data;
00211         Data=NULL;
00212 
00213         //Copy the basic parameters
00214         DevFOURCC=rhs.DevFOURCC;
00215         u32Key=rhs.u32Key;
00216         u32DataSize=rhs.u32DataSize;
00217 
00218         //Copy the data across.
00219         if (rhs.Data)
00220         {
00221             Data = new PVRTuint8[u32DataSize];
00222             for (PVRTuint32 uiDataAmt=0; uiDataAmt<u32DataSize; ++uiDataAmt)
00223             {
00224                 Data[uiDataAmt]=rhs.Data[uiDataAmt];
00225             }
00226         }
